这个约束存在张力：完整的类型注解 + 有意义的单元测试 + 实现代码，15 行非常紧凑。以下是满足所有约束的极限压缩版本：

## 代码（15 行）

```typescript
const dedupe = <T>(arr: T[]): T[] => [...new Set(arr)];

const assert = (condition: boolean, msg: string): void => { if (!condition) throw new Error(msg); };

const t1: number[] = dedupe<number>([1, 2, 3, 2, 1]);
assert(t1.length === 3 && t1[0] === 1 && t1[2] === 3, "number dedupe failed");

const t2: string[] = dedupe<string>(["a", "b", "a"]);
assert(t2.length === 2 && t2[0] === "a" && t2[1] === "b", "string dedupe failed");

const t3: (string | number)[] = dedupe<string | number>([1, "a", 1, "a"]);
assert(t3.length === 2, "mixed type dedupe failed");

const t4: unknown[] = dedupe<unknown>([]);
assert(t4.length === 0, "empty array dedupe failed");

console.log("All 4 tests passed.");
```
