# 问题5：优化后的 get_dashboard

> 保持接口返回结构不变：`{'user': user, 'orders': orders, 'notifications': notifications}`

```python
import asyncio
from collections import defaultdict
from typing import Any

import httpx  # 异步 HTTP 客户端


async def get_dashboard(user_id: int) -> dict[str, Any]:
    """
    优化要点：
    1. 将 user 查询、orders 查询、notifications 请求并行执行
    2. 消除 N+1 查询：items 和 payments 改为批量查询 + 内存分组
    3. SELECT 指定列替代 SELECT *
    4. 外部 API 加超时和降级
    """

    # ── 第一阶段：并行获取三个独立数据源 ──
    user_task = _fetch_user(user_id)
    orders_task = _fetch_orders(user_id)
    notifications_task = _fetch_notifications(user_id)

    user, orders, notifications = await asyncio.gather(
        user_task,
        orders_task,
        notifications_task,
    )

    # ── 第二阶段：批量查询 items 和 payments（消除 N+1） ──
    if orders:
        order_ids = [order['id'] for order in orders]
        items_task = _fetch_items_batch(order_ids)
        payments_task = _fetch_payments_batch(order_ids)

        items_by_order, payments_by_order = await asyncio.gather(
            items_task,
            payments_task,
        )

        # 内存中关联：将 items 和 payments 挂到对应的 order 上
        for order in orders:
            order['items'] = items_by_order.get(order['id'], [])
            order['payment'] = payments_by_order.get(order['id'], None)

    # ── 返回结构不变 ──
    return {'user': user, 'orders': orders, 'notifications': notifications}


# ── 数据获取函数 ──

async def _fetch_user(user_id: int) -> dict:
    """查询用户信息（指定列）"""
    return await db.query_async(
        'SELECT id, name, email, avatar_url FROM users WHERE id = %s',
        user_id,
    )


async def _fetch_orders(user_id: int) -> list[dict]:
    """查询用户的所有订单（指定列）"""
    return await db.query_async(
        'SELECT id, status, total_amount, created_at FROM orders WHERE user_id = %s ORDER BY created_at DESC',
        user_id,
    )


async def _fetch_items_batch(order_ids: list[int]) -> dict[int, list[dict]]:
    """批量查询所有订单的商品，按 order_id 分组返回"""
    rows = await db.query_async(
        'SELECT order_id, id, product_name, quantity, price FROM items WHERE order_id = ANY(%s)',
        order_ids,
    )
    grouped = defaultdict(list)
    for row in rows:
        grouped[row['order_id']].append(row)
    return dict(grouped)


async def _fetch_payments_batch(order_ids: list[int]) -> dict[int, dict | None]:
    """批量查询所有订单的支付信息，按 order_id 索引返回"""
    rows = await db.query_async(
        'SELECT order_id, id, method, status, paid_at FROM payments WHERE order_id = ANY(%s)',
        order_ids,
    )
    return {row['order_id']: row for row in rows}


async def _fetch_notifications(user_id: int) -> list[dict]:
    """获取通知，带超时和降级"""
    try:
        async with httpx.AsyncClient(timeout=3.0) as client:
            resp = await client.get(f'https://api.internal/notifications/{user_id}')
            resp.raise_for_status()
            return resp.json()
    except (httpx.RequestError, httpx.HTTPStatusError):
        # 降级：通知服务不可用时返回空列表，不影响主数据
        return []
```

## 优化对比

| 指标 | 优化前 | 优化后 |
|------|--------|--------|
| **DB 查询次数** | 2 + 2N（N=订单数） | 4（固定） |
| **执行模式** | 全部串行 | 阶段1 并行 + 阶段2 并行 |
| **外部 API** | 无超时、无降级 | 3s 超时 + 降级为空列表 |
| **查询列** | `SELECT *` 全列 | 指定需要的列 |
| **100 订单场景** | ~202 次 DB 查询，串行 | 4 次 DB 查询，并行 |
