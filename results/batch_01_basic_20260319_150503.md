-- 任务表
CREATE TABLE tasks (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    status VARCHAR(50) DEFAULT 'pending',  -- pending, in_progress, completed, blocked
    active_form VARCHAR(255),              -- TodoWrite的activeForm
    priority INTEGER DEFAULT 0,
    assignee_id INTEGER,
    created_by INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    estimated_hours DECIMAL,
    actual_hours DECIMAL,
    tags JSONB,
    metadata JSONB                        -- 额外字段
);

-- 任务上下文表
CREATE TABLE task_contexts (
    id SERIAL PRIMARY KEY,
    task_id INTEGER REFERENCES tasks(id) ON DELETE CASCADE,
    agent_session_id VARCHAR(255),        -- Agent会话ID
    context_type VARCHAR(50),             -- tool_call, response, todo_update
    context_data JSONB NOT NULL,           -- 序列化后的上下文数据
    timestamp TIMESTAMP DEFAULT NOW(),
    created_at TIMESTAMP DEFAULT NOW()
);

-- 评估批次表
CREATE TABLE evaluation_batches (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    agent_a_cmd VARCHAR(500),
    agent_b_cmd VARCHAR(500),
    total_questions INTEGER,
    status VARCHAR(50) DEFAULT 'pending',  -- pending, running, completed, failed
    results_file_path TEXT,
    created_by INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    started_at TIMESTAMP,
    completed_at TIMESTAMP,
    config JSONB                           -- 完整配置
);

-- 评估结果表
CREATE TABLE evaluation_results (
    id SERIAL PRIMARY KEY,
    batch_id INTEGER REFERENCES evaluation_batches(id) ON DELETE CASCADE,
    question_number INTEGER,
    question_text TEXT NOT NULL,
    agent_a_response TEXT,
    agent_b_response TEXT,
    agent_a_score JSONB,                   -- 评估分数
    agent_b_score JSONB,
    evaluator_comments TEXT,
    timestamp TIMESTAMP DEFAULT NOW()
);

-- 审核记录表
CREATE TABLE reviews (
    id SERIAL PRIMARY KEY,
    task_id INTEGER REFERENCES tasks(id) ON DELETE CASCADE,
    batch_id INTEGER REFERENCES evaluation_batches(id),
    reviewer_id INTEGER NOT NULL,
    status VARCHAR(50) DEFAULT 'pending',  -- pending, approved, rejected, needs_revision
    comment TEXT,
    reviewed_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,

    created_at TIMESTAMP DEFAULT NOW()

CREATE TABLE audit_logs (
