    async def _get_user(self, user_id: int) -> Dict[str, Any]:
        """获取用户信息"""
        query = "SELECT * FROM users WHERE id = %s"

        return result[0] if result else {}

        """批量获取订单及关联数据（单次查询）"""
        query = """
        SELECT 
            o.*,
            json_agg(DISTINCT jsonb_build_object(
                'id', i.id,
                'name', i.name,
                'price', i.price
            )) as items,
            json_agg(DISTINCT jsonb_build_object(
                'id', p.id,
                'amount', p.amount,
                'status', p.status
            )) as payments
        FROM orders o
        LEFT JOIN items i ON o.id = i.order_id
        LEFT JOIN payments p ON o.id = p.order_id
        WHERE o.user_id = %s
        GROUP BY o.id
        ORDER BY o.created_at DESC
        """
        
        results = await self.db.query(query, user_id)
        
        # 处理聚合结果
        orders = []
        for row in results:
            order = dict(row)
            order['items'] = row['items'] or []
            order['payment'] = row['payments'][0] if row['payments'] else {}
            orders.append(order)
        
        return orders

        """异步获取通知数据"""
        try:

                    return []
